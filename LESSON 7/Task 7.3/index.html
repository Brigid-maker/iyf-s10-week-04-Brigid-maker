// All five functions: function calculateArea(width, height) { return width * height; } function celsiusToFahrenheit(celsius) { return (celsius * 9/5) + 32; } function isEven(number) { return number % 2 === 0; } function getInitials(fullName) { return fullName.split(' ').map(word => word[0].toUpperCase()).join(''); } function reverseString(str) { return str.split('').reverse().join(''); }
