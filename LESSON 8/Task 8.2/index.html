Exercise 3 — Object.keys / .values / .entries
const scores = { math: 95, english: 88, science: 92 }; Object.keys(scores); // ["math", "english", "science"] Object.values(scores); // [95, 88, 92] Object.entries(scores); // [["math", 95], ["english", 88], ...] for (const [subject, score] of Object.entries(scores)) { console.log(`${subject}: ${score}`); }
